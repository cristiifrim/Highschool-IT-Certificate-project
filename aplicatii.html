Aplicația #1 - Crearea unei clase

În limbajul C++, clasele se pot creea cu tipul de date class. Intr-o clasă se pot defini două lucruri, variabilele utilizate ce sunt private si funcțiile specifice clasei, care doar ele pot accesa variabilele declarate anterior. Implementarea unei clase se poate face într-un fișier de tip 'header'(acesta se va include cu instrucțiunea '#include "nume_header.h"') sau direct în program. Pentru a vedea cum se creează o clasa în C++, click pe 'vezi sursa'.

General

Programarea orientată pe obiecte are peste un milion de aplicații în lumea reală. Exemplele anterioare sunt câteva exerciții pentru o bună înțelegere a conceptului de obiect. Prin folosirea datelor private, folosim Încapsularea. Când creăm un obiect, folosim Abstractizarea deoarece obiectul este abstract de celelalte. Prin definirea mai multor obiecte folosim Polimorfismul. Când se crează o clasă nouă prin care îi oferim atribute și metode dintr-o clasă părinte, folosim Moștenirea. Clasa nouă se numește și clasă derivată. 

      #include <iostream>
      #include <cstring>

      using namespace std;
      
      /*
          *Urmeaza definitia unei clase in C++
          *Aceasta se face folosind cuvantul class
          *Se vor defini datele private si cele publice
      
      */
      
      class Salut { //Definirea clasei
      
          
          /*
              *Acestea sunt datele private
          */
          private: 
              int varsta;
              char nume[101];
              char functie[25];
      
          public: //Datele publice ale clasei
          
          /*
              *Pentru a creea o un obiect, avem nevoie de o functie constructoare
              *Aceasta se apeleaza de fiecare data cand un obiecte este creat.
              *Cu aceasta functie setam valori implicite la variabilele clasei.
          */
      
              Salut(int x, char y[101], char z[25]) {
                  varsta = x;
                  strcpy(nume, y);
                  strcpy(functie, z);
              }
              int getAge() { // Aceasta functie ne ofera varsta persoanei cui ii spunem salut.
                  return varsta;
              }
              void Hello() { // Aceasta functie saluta omul pe baza varstei si functiei acestuia.
                  if(varsta > 30) {
                      cout << "Buna ziua, domnule " << functie << ' ' << nume << '.';
                  }
                  else {
                      if(strcmp(functie, "primar") != 0)
                          cout << "Salut, " << nume << '.';
                      else
                          cout << "Buna ziua tinere primar, " << nume << '.';
                  }
              }
      
      };
      
      int main()
      {
          /*
              *Urmeaza declararea unui obiect.
              *Acesta se va declara cu numarul de argumente ale functiei Constructoare
          */
          
          Salut sal(25, "Ifrim Cristian", "director"); //Crearea unui obiect nou
          sal.Hello(); //Aceasta functie apeleaza functia Hello din interiorul clasei.
      
          cout << sal.getAge(); // Aici afisam varsta obiectului nostru.
      
          /*
          Diferenta dintre declaratia noastra sal si clasa Salut este urmatoarea:
          Salut este clasa generala, fiind scrisa dupa cerintele noastre,
          iar sal este un obiect cu atribute specifice.
          
          Clasa este conceptul general, obiectele sunt membri ai claselor.
      
          */
          
          return 0;
      }
      
      #include <iostream>

      using namespace std;

      class Cont {

        private:
            int suma;
            float dobanda;
            string nume;
            string IBAN;
        public:
            Cont(int s, float i, string n, string iban);
            int showSum();
            float calculateInterest(int years);
            string showName();
            string showIBAN();
            void addSum(int x);
            void withdrawSum(int x);
      };

      Cont::Cont(int s, float i, string n, string iban) {
        suma = s;
        dobanda = i;
        nume = n;
        IBAN = iban;
        cout << "Utilizator creat:\nNume " << n << ' ' << "IBAN: " << iban << " Suma: " << s << " Dobanda: " << i << '\n';
      }

      int Cont::showSum() {
        return suma;
      }

      float Cont::calculateInterest(int years) {
        return dobanda*suma*years / 100;
      }

      string Cont::showName() {
        return nume;
      }

      string Cont::showIBAN() {
        return IBAN;
      }

      void Cont::addSum(int x) {
        if(x == 0) {
            cout << "Nu puteti adauga o suma nula de bani." << '\n';
        }
        else {
            suma += x;
            cout << "Ati adaugat cu succes suma de " << x << ". Sold total: " << suma << '.' << '\n';
        }
      }

      void Cont::withdrawSum(int x) {
        if(x < suma)
            suma -= x,
            cout << "Ati retras cu succes suma de " << x << ". Sold total: " << suma << '.' << '\n';
        else
            cout << "Suma invalida sau nula. Va rugam reincercati." << '\n';
      }

      int main()
      {
        Cont user(5000, 3.5, "Ifrim Cristian", "RO55RNCB0001");
        cout << user.calculateInterest(10) << '\n';
        user.addSum(2700);
        cout << user.showSum() << '\n';
        cout << user.calculateInterest(10) << '\n';
        user.withdrawSum(1500);
        cout << user.showIBAN() << '\n';
        return 0;
      }

        class Parcare {
            stack<int> liber;
            bool ocupat[51];
            Car masini[51];
            int timp[51];
            int costParcare = 5;

            public:
                Parcare();
                int getSpace() { return liber.size(); }
                void addCar(Car x);
                void removeCar(int place);
                void removeCar(string plate);
                int getCost(int place);
                int getCost(string plate);
                void carInfo(int place);
                void carInfo(string plate);

        };

        Parcare::Parcare() {
            for(int i = 50; i > 0; --i)
                liber.push(i);
            for(int i = 1; i <= 50; ++i)
                timp[i] = 0, ocupat[i] = 0;
        }

        void Parcare::addCar(Car x) {
            int ocupate = liber.top();
            masini[ocupate].setCar(x.getAge(), x.getName(), x.getModel(), x.getColor(), x.getFuel(), x.getVin(), x.getPlate());
            timp[ocupate] = 0;
            ocupat[ocupate] = 1;
            liber.pop();
            cout << "Masina parcata cu succes!\n";
        }

        void Parcare::removeCar(int place) {
            if(place != 0 && ocupat[place] == 1) {
                liber.push(place);
                ocupat[place] = 0;
                masini[place].setCar(0, "None", "None", "None", "None", "None", "None");
                cout << "Masina scoasa cu succes!\n";
                return;
            }
            cout << "Masina specifica nu a putut fi gasita sau locul este liber.\n";
        }

        void Parcare::removeCar(string plate) {

            for(int i = 1; i <= 50; ++i)
                if(masini[i].getPlate() == plate) {
                    liber.push(i);
                    ocupat[i] = 0;
                    timp[i] = 0;
                    masini[i].setCar(0, "None", "None", "None", "None", "None", "None");
                    cout << "Masina scoasa cu succes!\n";
                    return;
                }
            cout << "Masina specifica nu a putut fi gasita sau locul este liber.\n";
        }

        int Parcare::getCost(int place) {
            if(place != 0 && ocupat[place] == 1) {

                return timp[place] * costParcare;
            }
        }

        int Parcare::getCost(string plate) {
            for(int i = 1; i <= 50; ++i)
                if(masini[i].getPlate() == plate)
                    return timp[i] * costParcare;
        }

      #include <iostream>
      #include <cmath>

      using namespace std;

      class Complex {
          private:
              // i - partea imaginara, r - partea reala
              int i, r;
          public:
              // constructor stabileste valoare partii reale si acelei imaginare
              Complex(int pr, int pi);
              // constructor creeaza numarul complex nul
              Complex();
              // seteaza valorile partii reale si a celei imaginare
              void set(int pr, int pi);
              //returneaza modulul
              double modul();
              //returneaza partea reala
              int real();
              //returneaza partea imaginara
              int imag();
              //supraincarcare operator pentru afisarea numarului complex
              friend ostream& operator<<(ostream&, const Complex&);
              // supraincarcare operator adunare
              Complex operator +(Complex );
              // supraincarcare operator scadere
              Complex operator -(Complex );
              // supraincarcare operator inmultire
              Complex operator *(Complex );
      };

      Complex::Complex(int pr, int pi) {
          r = pr; i = pi;
      }
      Complex::Complex() {
          i = 0; r = 0;
      }
      void Complex::set(int pr, int pi) {
          r = pr; i = pi;
      }
      double Complex::modul() {
          return sqrt(i*i + r*r);
      }
      int Complex::real() {
          return r;
      }
      int Complex::imag() {
          return i;
      }
      ostream& operator<<(ostream& f, const Complex& b) {
          if(b.r == 0 && b.i == 0)
              return f << 0;
          if(b.r == 0)
              return f << b.i << 'i';
          if(b.i == 0)
              return f << b.r;
          if(b.i < 0)
              return f << b.r << b.i << 'i';
          return f << b.r << '+' << b.i << 'i';
      }
      // supraincarcare operator adunare
      Complex Complex::operator +(Complex b) {
          Complex temp = *this;
          temp.r += b.r;
          temp.i += b.i;
          return temp;
      }
      // supraincarcare operator scadere
      Complex Complex::operator -(Complex b) {
          Complex temp = *this;
          temp.r -= b.r;
          temp.i -= b.i;
          return temp;
      }
      // supraincarcare operator inmultire
      Complex Complex::operator *(Complex b) {
          Complex temp = *this;
          temp.r = this->r * b.r - this->i*b.i;
          temp.i = this->r * b.i + this->i * b.r;
          return temp;
      }
